# something goes wrong and an error message is produced.

# paying the cost of redetecting them again in ExternalProject_Add()

  [==[C:/Program Files/Git/cmd/git.exe;2.27.0.windows.1]==]
)

                     "UPDATE_DISCONNECTED" "False" "GIT_REPOSITORY" "https://github.com/saleae/AnalyzerSDK.git" "GIT_TAG" "master" "GIT_SHALLOW" "True" "GIT_PROGRESS" "True"
                    SOURCE_DIR          "C:/languages/c/nled-async-pixel-analyzer/out/build/x64-Release/_deps/analyzersdk-src"
                    BINARY_DIR          "C:/languages/c/nled-async-pixel-analyzer/out/build/x64-Release/_deps/analyzersdk-build"
                    CONFIGURE_COMMAND   ""
                    BUILD_COMMAND       ""
                    INSTALL_COMMAND     ""
                    TEST_COMMAND        ""
                    USES_TERMINAL_DOWNLOAD  YES
                    USES_TERMINAL_UPDATE    YES
                    USES_TERMINAL_PATCH     YES
)
